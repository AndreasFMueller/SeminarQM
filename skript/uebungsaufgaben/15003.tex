z^3

\cos^3\varphi + 3i\cos^2\varphi\sin\varphi - 3 \cos\varphi\sin^2\varphi
-i\sin^3\varphi
\\
z^3
&=
e^{3i\varphi}=\cos3\varphi+i\sin3\varphi
\end{align*}

\begin{align*}

\cos3\varphi

3\cos^2\varphi\sin\varphi-\sin^3\varphi

\sin3\varphi

&=
\cos3\varphi

\sin3\varphi

&=
\cos3\varphi
&
3\sin\varphi -\sin^3\varphi-\sin^3\varphi

\sin3\varphi

4\cos^3\varphi
&=
\cos3\varphi

3\cos\varphi

\sin3\varphi

\cos^3\varphi

\end{loesung}
